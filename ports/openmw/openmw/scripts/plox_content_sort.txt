#!/usr/bin/env python3
#
# TITLE: plox Content Sorter
# AUTHOR: Gemini 2.5 Pro
# DESCRIPTION: Sorts content files (plugins) using the external plox tool.
# ARGS: --config-dir "$OPENMW_DIR" --openmw-config "$OPENMW_CFG"
# REGISTER: SORT_CONTENT
# PRIORITY: 1000
# HAS_OUTPUT: TRUE
#

import argparse
import os
import sys
import subprocess
import json


def find_plox_executable():
    """Finds the plox executable in common locations."""
    # Look in the same directory as this script first
    script_dir = os.path.dirname(os.path.realpath(__file__))
    if os.path.exists(os.path.join(script_dir, "plox")):
        return os.path.join(script_dir, "plox")
    # Look in PATH
    for path in os.environ.get("PATH", "").split(os.pathsep):
        if os.path.exists(os.path.join(path, "plox")):
            return "plox"
    return None


def main():
    parser = argparse.ArgumentParser(description="Sorts content files in openmw.cfg using plox.")
    parser.add_argument('--openmw-config', required=True, help="Path to openmw.cfg")
    parser.add_argument('--config-dir', required=True, help="Path to store plox data files.")
    parser.add_argument('--dry-run', action='store_true', help="Print the new order without modifying the file.")
    args = parser.parse_args()

    plox_exec = find_plox_executable()
    if not plox_exec:
        print("ESMM::ALERT::Error::plox executable not found in PATH or scripts directory.")
        sys.exit(1)

    if not os.path.exists(args.openmw_config):
        print(f"ESMM::ALERT::Error::Config file not found: {args.openmw_config}")
        sys.exit(1)

    # We pass 'y' to stdin to auto-accept any prompts plox might have
    command = [plox_exec, "sort", "-r", args.config_dir, "-c", args.openmw_config, "-d"]

    if (    os.path.exists(os.path.join(args.config_dir, 'mlox_base.txt'))
        and os.path.exists(os.path.join(args.config_dir, 'mlox_user.txt'))):
        command.append('-n')
    
    print(f"Running command: {' '.join(command)}")

    try:
        process = subprocess.run(command, input='y\n', text=True, capture_output=True, check=False)

        # Print plox's own output for logging/debugging
        print("\n--- plox stdout ---\n" + process.stdout)
        if process.stderr:
            print("\n--- plox stderr ---\n" + process.stderr)

        if process.returncode != 0:
            print(f"\nESMM::ALERT::Error::plox exited with a non-zero status code: {process.returncode}")
            sys.exit(process.returncode)

        # Find the "New order:" line and parse the JSON-like array
        new_order_line = None
        for line in '\n'.join((process.stdout, process.stderr)).splitlines():
            if "New order:" in line:
                new_order_line = ""
            elif new_order_line is None:
                continue
            elif new_order_line == "":
                new_order_line = line.strip()
            else:
                break
        
        if not new_order_line:
            # If plox didn't provide a new order, it might mean nothing changed, which is fine.
            print("plox did not return a new order. Assuming order is already correct.")
            sys.exit(0)

        # Extract the list part of the line
        list_str = new_order_line[new_order_line.find('['):]
        sorted_plugins = json.loads(list_str)

        if args.dry_run:
            print("\n--- Dry Run: New Content File Order (from plox) ---")
            for plugin in sorted_plugins:
                print(f"content=\"{plugin}\"")
            sys.exit(0)

        # Read the original file to get comments and disabled status
        with open(args.openmw_config, 'r', encoding='utf-8') as f:
            original_lines = f.readlines()

        # Create a map of disabled plugins
        disabled_plugins = set()
        for line in original_lines:
            stripped = line.strip()
            if stripped.lower().startswith('#content='):
                plugin_name = stripped.split('=', 1)[1].strip().strip('"')
                disabled_plugins.add(plugin_name.lower())

        with open(args.openmw_config, 'w', encoding='utf-8') as f:
            content_written = False
            for line in original_lines:
                stripped = line.strip()
                if not (stripped.lower().startswith('content=') or stripped.lower().startswith('#content=')):
                    f.write(line)
                elif not content_written:
                    for plugin in sorted_plugins:
                        prefix = "#" if plugin.lower() in disabled_plugins else ""
                        f.write(f"{prefix}content=\"{plugin}\"\n")
                    content_written = True
        
        print("\nSuccessfully sorted content files using plox and saved openmw.cfg.")

    except FileNotFoundError:
        print("ESMM::ALERT::Error::plox command not found. Is it installed and in your PATH?")
        sys.exit(1)
    except json.JSONDecodeError:
        print("ESMM::ALERT::Error::Could not parse the 'New order:' line from plox output.")
        sys.exit(1)
    except Exception as e:
        print(f"ESMM::ALERT::Error::An unexpected error occurred: {e}")
        sys.exit(1)

if __name__ == '__main__':
    main()
